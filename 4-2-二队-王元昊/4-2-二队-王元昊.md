# weird-py

得到一个脚本

```py
import ast

FLAG = "shellmates{XXXXXXXXXXXXXXXXXXXXXXXXXX}"

if __name__ == '__main__':
	try:
		a = ast.literal_eval(input("Enter a: "))
		assert isinstance(a, (int,float)), True//已经提示是浮点
		b = ast.literal_eval(input("Enter b: "))
		assert isinstance(b, (int,float)), True//已经提示是浮点了
	except://a和b必须是数字
		print("Only numbers are allowed")
		exit()

	if (a/b).is_integer() and (a%b != 0)://就是a/b是整数，a%b不等于0
		print(FLAG)//不是数字就会退出
	else:
		print("Try again!")

```
